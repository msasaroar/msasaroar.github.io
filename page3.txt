    <script>
        function loadContent(url) {
            fetch(url)
                .then(response => response.text())
                .then(data => {
                    document.getElementById('content-display').innerHTML = data;
                })
                .catch(error => {
                    document.getElementById('content-display').innerHTML = "Error loading content.";
                });
        }
    </script>
